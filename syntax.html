Input / Output Functions

🟢 Beginner

printf()

Purpose: Prints formatted output to screen.

printf("Hello %s, age %d\n", name, age);

❌ Forgetting \n when a newline is needed.

scanf()

Purpose: Reads formatted input from keyboard.

int age;
scanf("%d", &age);   // & is mandatory!

❌ Missing & before variables.

getchar()

Purpose: Reads a single character from stdin.

char ch = getchar();

putchar()

Purpose: Prints a single character to screen.

putchar('A');

gets() ⚠ deprecated

Purpose: Reads a line (unsafe — removed in C11).

char s[100];
gets(s);   // avoid

puts()

Purpose: Prints a string + newline.

puts("Hello World");

Decision Making

🟢 Beginner

if Statement

if (age >= 18) {
    printf("Eligible to vote");
}

if-else

if (marks >= 40)
    printf("Pass");
else
    printf("Fail");

Nested if

if (marks >= 90)       grade = 'A';
else if (marks >= 75)  grade = 'B';
else if (marks >= 60)  grade = 'C';
else                   grade = 'D';

switch

switch (choice) {
    case 1: printf("Add"); break;
    case 2: printf("Sub"); break;
    default: printf("Invalid");
}
Use cases: if for ranges, switch for fixed discrete values. Always include break in switch to prevent fall-through.
08

Loops

🟢 Beginner

for Loop

for (int i = 0; i < 5; i++) {
    printf("%d ", i);
}

while Loop

int i = 0;
while (i < 5) {
    printf("%d ", i);
    i++;
}

do-while Loop

int i = 0;
do {
    printf("%d ", i);
    i++;
} while (i < 5);   // runs at least once

break & continue

for (int i = 0; i < 10; i++) {
    if (i == 5)  break;     // exit loop
    if (i % 2)   continue;  // skip odd
    printf("%d ", i);
}

Comparison

Functions

🟡 Intermediate

Declaration / Definition / Call

#include <stdio.h>

// 1. Declaration (prototype)
int add(int a, int b);

int main() {
    int result = add(5, 3);   // 3. Call
    printf("%d", result);     // 8
    return 0;
}

// 2. Definition
int add(int a, int b) {
    return a + b;
}

Recursion

int factorial(int n) {
    if (n == 0) return 1;          // base case
    return n * factorial(n - 1);   // recursive call
}
Tip: Always provide a base case in recursion to avoid stack overflow.

Pointers

🔴 Advanced

Declaration & Dereference

int x = 10;
int *p = &x;      // p holds address of x

printf("%d",  x);   // value: 10
printf("%p",  p);   // address
printf("%d", *p);   // dereference: 10

Pointer Arithmetic

int arr[3] = {10, 20, 30};
int *p = arr;
printf("%d", *p);       // 10
printf("%d", *(p+1));   // 20
printf("%d", *(p+2));   // 30

Call by Reference

void swap(int *a, int *b) {
    int t = *a; *a = *b; *b = t;
}

int main() {
    int x = 5, y = 10;
    swap(&x, &y);   // x=10, y=5
    return 0;
}

Structures

🟡 Intermediate

Declaration & Access

struct Student {
    char name[30];
    int roll;
    float marks;
};

int main() {
    struct Student s1 = {"Shyam", 42, 88.5};

}

struct Student class[3] = {
    {"A", 1, 80},
    {"B", 2, 75},
    {"C", 3, 90}

for (int i = 0; i < 3; i++)

Nested Structure

struct Person {

Unions

🟡 Intermediate

Like struct, but all members share the same memory. Size = size of largest member.

#include <stdio.h>
union Data {
    int   i;
    float f;
    char  str[20];
};

int main() {
    union Data d;
    d.i = 10;
    printf("%d\n", d.i);
    d.f = 3.14;       // overwrites i
    printf("%.2f\n", d.f);
    return 0;
}

File Handling

🟡 Intermediate
FunctionPurpose
fopen(name, mode)Open a file
fclose(fp)Close a file
fprintf(fp, ...)Write formatted data
fscanf(fp, ...)Read formatted data
fgets(buf, n, fp)Read a line
fputs(s, fp)Write a string
fread/fwriteBinary I/O

Modes: r w a r+ w+ a+ rb wb

#include <stdio.h>
int main() {
    FILE *fp = fopen("data.txt", "w");
    if (fp == NULL) return 1;

    fprintf(fp, "Hello File!\n");
    fputs("Second line\n", fp);
    fclose(fp);

    fp = fopen("data.txt", "r");
    char line[100];
    while (fgets(line, sizeof(line), fp))
        printf("%s", line);
    fclose(fp);
    return 0;
}

Dynamic Memory Allocation

🔴 Advanced
FunctionPurpose
malloc(size)Allocates uninitialized memory
calloc(n, size)Allocates zero-initialized memory
realloc(ptr, size)Resizes a previous allocation
free(ptr)Releases memory
 
#include <stdio.h>
#include <stdlib.h>
int main() {
    int *p = (int *)malloc(5 * sizeof(int));
    if (p == NULL) return 1;

    for (int i = 0; i < 5; i++) p[i] = i + 1;

    p = (int *)realloc(p, 10 * sizeof(int));

    free(p);   // ALWAYS free!
    return 0;
}
⚠ Common Errors:
  • Forgetting to free() → memory leak.
  • Using memory after free() → dangling pointer.
  • Not checking for NULL after allocation.

Common Errors & Fixes

🟢 Beginner
❌ Missing Semicolon

Cause: Every C statement must end with ;

printf("Hi")    // ❌
printf("Hi");   // ✅
❌ Infinite Loop

Cause: Loop condition never becomes false.

int i = 0;
while (i < 5) {       // missing i++
    printf("%d", i);
}
❌ Wrong Format Specifier

Cause: Mismatch between type and format.

float x = 3.14;
printf("%d", x);   // ❌ garbage
printf("%f", x);   // ✅
❌ Array Out Of Bounds

Cause: Accessing index outside declared range.

int a[5];
a[5] = 10;   // ❌ valid indices: 0-4
❌ Uninitialized Variable

Cause: Reading a variable before assigning.

int x;
printf("%d", x);   // ❌ undefined
int x = 0; printf("%d", x);   // ✅
❌ Null Pointer Access

Cause: Dereferencing a NULL pointer.

int *p = NULL;
*p = 5;          // ❌ crash
if (p) *p = 5;   // ✅
❌ Missing Header File

Cause: Using a function without including its header.

printf("Hi");   // ❌ if no #include <stdio.h>
#include <stdio.h>   // ✅
